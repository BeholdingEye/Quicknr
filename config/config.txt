#                           QLM OR MARKDOWN                            #
#                                                                      #
#  Quicknr features its own Light Markup syntax (QLM) for easy plain   #
#  text formatting of web page ".txt" source files.                    #
#                                                                      #
#  If you prefer Markdown and have the Markdown module installed, you  #
#  may switch to it with the following setting. Valid values are QLM   #
#  and MARKDOWN, QLM being the default.                                #
#                                                                      #
#  Regardless of this setting, if the Markdown module is available,    #
#  Markdown will be used for conversion of source files with the       #
#  ".mdml" file extension. This way you can have both: with QLM as     #

#  ".mdml" sources with the Markdown module.                           #
#                                                                      #
#  Markdown support is experimental: it is not supported in news       #
#  pages, no reliable HTML titling is done by Quicknr and the Quicknr  #
#  classes are not present.                                            #
#                                                                      #
#  A compromise titling solution is offered for Markdown sources: up   #
#  to 80 characters of the first paragraph - without indentation - in  #
#  the source file are read as the HTML page title. This can be        #
#  turned off if MARKDOWN_TITLING is set to NO. The default is YES.    #
#                                                                      #
#  If you like the Markdown titling feature but would want to turn it  #
#  off in some files, type one or more whitespace characters at the    #
#  very start of the file: the titling feature expects no whitespace   #
#  there. Then the page title will be the setting from "config.txt".   #
#                                                                      #

########################################################################
#                                                                      #
#                           NEWS LIST THUMBS                           #
#                                                                      #
#  This setting controls whether Quicknr will prepend image            #
#  thumbnails to news post intro blurbs in the news list page. It's a  #
#  nice feature to have, adding visual candy to what could otherwise   #
#  appear as quite a dry textual page.                                 #
#                                                                      #
#  By default the feature is on with the value of YES. Disable with    #
#  NO.                                                                 #
#                                                                      #
#  Quicknr will use the first image it finds in the source news file   #
#  for the thumbnail.                                                  #
#                                                                      #
#  Rather than use the full-size images, Quicknr will look for their   #
#  thumbnail counterparts, in the same directory, if the image link    #
#  is to a local file on the server. Thumbnails should have the same   #
#  name as the full-size file, but ending with "thumb.jpg".            #
#                                                                      #
#  For example, if the image in your news post is "dog.jpg", the       #
#  thumbnail should be "dog-thumb.jpg", or "dog_thumb.jpg" or          #
#  "dogthumb.jpg".                                                     #
#                                                                      #
#  If thumbnails aren't found, and the Python Imaging Library (PIL)    #
#  module is installed (it isn't by default), Quicknr will create a    #
#  thumbnail of the original image and use it.                         #
#                                                                      #
#  If thumbnails aren't found and PIL is not available, the full-size  #
#  images will be used. In this case, it helps if the images are       #
#  saved as progressive JPEGs, so that at least a low res version is   #
#  shown before the image loads completely.                            #
#                                                                      #
#  NEWS_LIST_THUMB_SIZE sets the pixel size of the rectangular thumb   #
#  that will be created. Default value is 100.                         #
#                                                                      #
NEWS_LIST_THUMBS: YES
NEWS_LIST_THUMB_SIZE: 100

#                            =============                             #
#                                                                      #
#  Add your code in either the HTML_HEAD or HTML_TAIL section,         #
#  anywhere that is appropriate. Processed content of the text source  #
#  document will be placed between these two snippets to create the    #
#  HTML file.                                                          #
#                                                                      #
#  Quicknr creates tags in XHTML format, compatible with XHTML 1.0     #
#  and HTML 5. The default snippets below are XHTML 1.0, but you can   #
#  change that. Note that Quicknr is designed for publishing websites, #
#  not web apps.                                                       #
#                                                                      #
#  When editing these snippets, be careful to not remove the  """      #
#  quotes around the snippet.                                          #
#                                                                      #
#                                 PHP                                  #
#                                                                      #
#  PHP code can be included. So instead of this:                       #
#                                                                      #
#      <p>                                                             #
#        Hello world!                                                  #
#      </p>                                                            #
#                                                                      #
#  ...you can do this:                                                 #
#                                                                      #
#      <?php echo '<p>Hello world!</p>'; ?>                            #
#                                                                      #
#  ...provided that the server is configured for PHP service.          #
#                                                                      #
#  If you use PHP, be sure to set the PAGE_FILE_EXTENSION preference   #
#  to ".php".                                                          #
#                                                                      #
#                               IMPORTS                                #
#                                                                      #
#  Imports are supported in the following format:                      #
#                                                                      #
#  @import: "file-to-import"                                           #
#                                                                      #
#  The import directive will be replaced by the contents of the file   #
#  being imported. The file should reside in the "config/import/"      #
#  directory of the website folder.                                    #
#                                                                      #
#  Import files must follow a naming convention: "all.xxx" will        #
#  import into all conversions (the file extension can be anything,    #
#  ".txt", ".html", etc.). Other import names will import only into    #
#  files with a matching name and any of the acceptable file           #
#  extensions.                                                         #
#                                                                      #
#  For example, an import file named "about.txt" will import into      #
#  HTML conversions of sources named "about.txt", "about.mdml",        #
#  "about.html", etc., but no others.                                  #
#                                                                      #
#  Conversely, this means that an import file named "something.txt"    #
#  will not import into anything if a "something.xxx" source file      #
#  does not exist.                                                     #
#                                                                      #
#  To enable multiple imports into a file, import file names can be    #
#  prefixed with an ASCII alphanumerical string and double underscore. #
#  The underscore is doubled to avoid conflict with source names       #
#  containing single underscores.                                      #
#                                                                      #
#  So you might use "h__about.html" to import into the head snippet    #
#  of "about.xxx" sources, and "t__about.html" for the tail. For more  #
#  detail, try "metas__about.html", "csslink__about.html",             #
#  "footer1__about.html", etc.                                         #
#                                                                      #
#  Finally, a file named "newspost.txt" (optionally with a prefix as   #
#  detailed above) will import only into sources that reside in the    #
#  "page_sources/news" directory of the website, regardless of their   #
#  names.                                                              #
#                                                                      #
#  If more conditional granularity is required, Python functions       #
#  should be used instead.                                             #
#                                                                      #
#  Recursive imports are supported; import files may contain import    #
#  directives, as well as Python directives.                           #
#                                                                      #
#  By way of example, an import statement is part of the default       #
#  HTML_HEAD snippet. The import does nothing, the "all.txt" file is   #
#  empty except for a comment.                                         #
#                                                                      #
#                                PYTHON                                #
#                                                                      #
#  Python code is supported with function (or class) calls:            #
#                                                                      #
#  @python: "name-of-function"                                         #
#                                                                      #
#  Such a Python directive calls a function defined in the             #
#  "config/user_functions.py" file of the website, to work on and      #
#  output its result into completed HTML text of a source page just    #
#  before it is written to file.                                       #
#                                                                      #
#  In addition to their placement in snippets, Python directives       #
#  included in source text documents will work, as will import         #
#  directives.                                                         #
#                                                                      #
#  The function must accept exactly three positional parameters:       #
#                                                                      #
#  1) Text of the full HTML output just before writing to file         #
#                                                                      #
#  2) Starting index position of the function directive in the text    #
#                                                                      #
#  3) Configuration dictionary holding website settings. Keys of the   #
#     dictionary have the same names as options in the "config.txt"    #
#     file. Some additional keys are also present: "siteDir" for       #
#     website folder path, "siteFolder" for its folder name, and the   #
#     self-explanatory "sourceFilePath" and "htmlFilePath" of the      #
#     file being worked on. The dictionary is read-only.               #
#                                                                      #
#  The function directive is removed from the text before the call is  #
#  made. After it has done its work, the function must return the      #
#  processed text.                                                     #
#                                                                      #
#  Note that all importing is done before any of the functions are     #
#  called. Functions are then called in sequence from top to bottom    #
#  of the text. While the function directive of the function being     #
#  called is removed, subsequent function directives remain in the     #
#  text until it is their turn to be called. If an earlier function    #
#  removes a later function directive from the text, the later         #
#  function will not be called.                                        #
#                                                                      #
#  One example function is defined in the Quicknr "user_functions.py"  #
#  file: "page_style_link", importing matching sylesheet links into    #
#  some pages. Adapt it to your own needs.                             #
#                                                                      #
#                               ESCAPES                                #
#                                                                      #
#  If you need the literal "@import:" or "@python:" text in a HTML     #
#  snippet or file, use the char reference "&#64;" instead of "@".     #
#                                                                      #
########################################################################
